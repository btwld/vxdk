## Usage


```html
<!-- Element to attach VXDK -->
<div id="vxdk"></div>

<script>
  const options = {
    source: "https://url-to/video",
    preload: "metadata",
    loop: false,
  };

  document.addEventListener("DOMContentLoaded", () => {
    const el = document.getElementById("vxdk");
    const vxdk = Vxdk.init(el, options);
  });
</script>
```
